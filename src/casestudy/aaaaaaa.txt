    public void upperDiamond(int a) {
        for (int i = 0; i < a; i++) {
            for (int j = a - 1; j > i; j--)
                System.out.print(" ");
            for (int k = 0; k < 2 * i - 1; k++)
                System.out.print("*");
            System.out.print("\n");
        }
    }

    public void lowerDiamond(int b) {
        for (int i = 0; i < b; i++) {
            for (int j = 0; j <= i; j++)
                System.out.print(" ");
            for (int k = 0; k < 2 * (b - i) - 1; k++)
                System.out.print("*");
            System.out.print("\n");
            
            
        }
